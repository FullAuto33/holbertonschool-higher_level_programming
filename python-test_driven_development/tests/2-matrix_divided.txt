matrix_divided = __import__('2-matrix_divided').matrix_divided

# Test correct
matrix1 = [[1, 2, 3], [4, 5, 6]]
print(matrix_divided(matrix1, 3))
# Résultat attendu : [[0.33, 0.67, 1.0], [1.33, 1.67, 2.0]]

# Test : mauvaise matrice (pas une liste de listes)
try:
    matrix_divided([1, 2, 3], 3)
except Exception as e:
    print(e)  # Expected: matrix must be a matrix (list of lists) of integers/floats

# Test : diviseur est 0
try:
    matrix_divided([[1, 2], [3, 4]], 0)
except Exception as e:
    print(e)  # Expected: division by zero

# Test : lignes de tailles différentes
try:
    matrix_divided([[1, 2], [3, 4, 5]], 2)
except Exception as e:
    print(e)  # Expected: Each row of the matrix must have the same size

# Test : div n'est pas un nombre
try:
    matrix_divided([[1, 2], [3, 4]], "a")
except Exception as e:
    print(e)  # Expected: div must be a number
